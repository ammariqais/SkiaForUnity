24 public static void Parse(TextBlock tb, string text, Style baseStyle) {
25 if (string.IsNullOrEmpty(text)) return;
26
27 var stack = new Stack<TagState>();
28 var current = new TagState {
29 bold = baseStyle.FontItalic ? false : false, // will read from base
30 italic = baseStyle.FontItalic,
31 underline = baseStyle.Underline,
32 strikeThrough = baseStyle.StrikeThrough,
33 variant = baseStyle.FontVariant,
34 fontSize = baseStyle.FontSize,
35 textColor = baseStyle.TextColor,
36 hasColor = false
37 };
38 current.bold = baseStyle.FontWeight >= 700;
39 current.italic = baseStyle.FontItalic;
40
41 int i = 0;
42 int runStart = 0;
43
44 while (i < text.Length) {
45 if (text[i] == '<') {
46 int closeIdx = text.IndexOf('>', i + 1);
47 if (closeIdx == -1) {
48 i++;
49 continue;
50 }
51
52 string tagContent = text.Substring(i + 1, closeIdx - i - 1).Trim();
53 string tagLower = tagContent.ToLowerInvariant();
54
55 bool isKnownTag = true;
56 bool isClosing = tagLower.StartsWith("/");
57 string tagName = isClosing ? tagLower.Substring(1) : tagLower;
58
59 // Check for value tags like size=24 or color=#FF0000
60 string tagValue = null;
61 int eqIdx = tagName.IndexOf('=');
62 if (eqIdx >= 0) {
63 tagValue = tagName.Substring(eqIdx + 1);
64 tagName = tagName.Substring(0, eqIdx);
65 }
66
67 switch (tagName) {
68 case "b":
69 case "i":
70 case "u":
71 case "s":
72 case "sup":
73 case "sub":
74 case "size":
75 case "color":
76 break;
77 default:
78 isKnownTag = false;
79 break;
80 }
81
82 if (!isKnownTag) {
83 i++;
84 continue;
85 }
86
87 // Flush text before this tag
88 if (i > runStart) {
89 string segment = text.Substring(runStart, i - runStart);
90 AddStyledRun(tb, segment, baseStyle, current);
91 }
92
93 if (isClosing) {
94 // Pop state
95 if (stack.Count > 0) {
96 current = stack.Pop();
97 }
98 } else {
99 // Push current state
100 stack.Push(current);
101
102 switch (tagName) {
103 case "b":
104 current.bold = true;
105 break;
106 case "i":
107 current.italic = true;
108 break;
109 case "u":
110 current.underline = UnderlineStyle.Solid;
111 break;
112 case "s":
113 current.strikeThrough = StrikeThroughStyle.Solid;
114 break;
115 case "sup":
116 current.variant = FontVariant.SuperScript;
117 break;
118 case "sub":
119 current.variant = FontVariant.SubScript;
120 break;
121 case "size":
122 if (tagValue != null && float.TryParse(tagValue, System.Globalization.NumberStyles.Float, System.Globalization.CultureInfo.InvariantCulture, out float sz)) {
123 current.fontSize = sz;
124 }
125 break;
126 case "color":
127 if (tagValue != null) {
128 if (TryParseColor(tagValue, out SKColor c)) {
129 current.textColor = c;
130 current.hasColor = true;
131 }
132 }
133 break;
134 }
135 }
136
137 runStart = closeIdx + 1;
138 i = runStart;
139 } else {
140 i++;
141 }
142 }
143
144 // Flush remaining text
145 if (runStart < text.Length) {
146 string segment = text.Substring(runStart);
147 AddStyledRun(tb, segment, baseStyle, current);
148 }
149 }

151 private static void AddStyledRun(TextBlock tb, string text, Style baseStyle, TagState state) {
152 if (string.IsNullOrEmpty(text)) return;
153
154 var style = baseStyle.Modify(
155 fontWeight: state.bold ? 700 : (int?)baseStyle.FontWeight,
156 fontItalic: state.italic,
157 underline: state.underline,
158 strikeThrough: state.strikeThrough,
159 fontVariant: state.variant,
160 fontSize: state.fontSize,
161 textColor: state.hasColor ? state.textColor : (SKColor?)null
162 );
163
164 tb.AddText(text, style);

167 private static bool TryParseColor(string value, out SKColor color) {
168 color = default;
169
170 if (value.StartsWith("#")) {
171 string hex = value.Substring(1);
172 if (hex.Length == 6) {
173 if (uint.TryParse(hex, System.Globalization.NumberStyles.HexNumber, null, out uint rgb)) {
174 color = new SKColor((byte)((rgb >> 16) & 0xFF), (byte)((rgb >> 8) & 0xFF), (byte)(rgb & 0xFF));
175 return true;
176 }
177 } else if (hex.Length == 8) {
178 if (uint.TryParse(hex, System.Globalization.NumberStyles.HexNumber, null, out uint rgba)) {
179 color = new SKColor((byte)((rgba >> 24) & 0xFF), (byte)((rgba >> 16) & 0xFF), (byte)((rgba >> 8) & 0xFF), (byte)(rgba & 0xFF));
180 return true;
181 }
182 }
183 }
184
185 // Named colors
186 switch (value.ToLowerInvariant()) {
187 case "red": color = new SKColor(255, 0, 0); return true;
188 case "green": color = new SKColor(0, 128, 0); return true;
189 case "blue": color = new SKColor(0, 0, 255); return true;
190 case "white": color = new SKColor(255, 255, 255); return true;
191 case "black": color = new SKColor(0, 0, 0); return true;
192 case "yellow": color = new SKColor(255, 255, 0); return true;
193 case "cyan": color = new SKColor(0, 255, 255); return true;
194 case "magenta": color = new SKColor(255, 0, 255); return true;
195 case "orange": color = new SKColor(255, 165, 0); return true;
196 case "grey": case "gray": color = new SKColor(128, 128, 128); return true;
197 }
198
199 return false;
200 }
